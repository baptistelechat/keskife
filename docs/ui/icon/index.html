Génération des assets PWA

# 1. Installer la lib (companion officiel de vite-plugin-pwa)
pnpm add -D @vite-pwa/assets-generator

# 2. Créer pwa-assets.config.ts à la racine
import { defineConfig, minimalPreset } from '@vite-pwa/assets-generator/config'

export default defineConfig({
  preset: minimalPreset,       // génère 64/192/512 + apple-touch + maskable
  images: ['public/logo.svg'], // source SVG 512×512 sans padding
})

# 3. Ajouter dans package.json → scripts
"generate-pwa-assets": "pwa-assets-generator"

# 4. Exporter la variante choisie en SVG dans public/logo.svg puis lancer
pnpm generate-pwa-assets
# → public/pwa-64x64.png, pwa-192x192.png, pwa-512x512.png
#    apple-touch-icon-180x180.png, maskable-icon-512x512.png
